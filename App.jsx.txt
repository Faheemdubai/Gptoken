                      <th key={i} style={{ padding: "10px 16px", fontSize: 12, color: "#8a9bb8", fontWeight: 600, textAlign: isUrdu ? "right" : "left" }}>{h}</th>
                    ))}
                  </tr>
                </thead>
                <tbody>
                  {portfolioData.map(p => (
                    <tr key={p.id} style={{ borderBottom: "1px solid rgba(255,255,255,0.04)" }}>
                      <td style={{ padding: "16px", fontWeight: 600 }}>{isUrdu ? p.nameUr : p.name}</td>
                      <td style={{ padding: "16px", color: "#c9a84c" }}>{p.tokens}</td>
                      <td style={{ padding: "16px" }}>? {fmt(p.value)}</td>
                      <td style={{ padding: "16px" }}><span style={{ color: "#4ade80" }}>+{p.return}%</span></td>
                      <td style={{ padding: "16px" }}><button className="outline-btn" style={{ padding: "6px 14px", fontSize: 12, fontFamily: "inherit" }}>{t.marketplace.sell}</button></td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
          </div>
        )}
      </main>

      {/* PAYMENT MODAL */}
      {showPayment && selectedProperty && (
        <div className="modal-overlay" onClick={() => setShowPayment(false)}>
          <div className="modal" onClick={e => e.stopPropagation()}>
            {paymentSuccess ? (
              <div className="success-anim" style={{ textAlign: "center", padding: "20px 0" }}>
                <div style={{ fontSize: 60, marginBottom: 16 }}>?</div>
                <h2 style={{ color: "#4ade80", fontSize: 22, marginBottom: 8 }}>{isUrdu ? "??????? ??????!" : "Payment Successful!"}</h2>
                <p style={{ color: "#8a9bb8" }}>{isUrdu ? "????? ?? ?? ?????? ??? ???? ?? ???" : "Tokens added to your portfolio"}</p>
              </div>
            ) : (
              <>
                <h2 style={{ fontSize: 20, fontWeight: 700, color: "#c9a84c", marginBottom: 4 }}>{t.payment.title}</h2>
                <p style={{ color: "#8a9bb8", fontSize: 13, marginBottom: 24 }}>{isUrdu ? selectedProperty.nameUr : selectedProperty.name}</p>
                <div style={{ marginBottom: 18 }}>
                  <label style={{ fontSize: 13, color: "#8a9bb8", display: "block", marginBottom: 8 }}>{isUrdu ? "????? ?? ?????" : "Number of Tokens"}</label>
                  <input className="input-field" type="number" min={1} max={selectedProperty.totalTokens - selectedProperty.soldTokens} value={tokenAmount} onChange={e => setTokenAmount(Math.max(1, parseInt(e.target.value) || 1))} />
                </div>
                <div style={{ background: "rgba(201,168,76,0.06)", border: "1px solid rgba(201,168,76,0.15)", borderRadius: 10, padding: 14, marginBottom: 18 }}>
                  <div style={{ display: "flex", justifyContent: "space-between", marginBottom: 8, fontSize: 13 }}>
                    <span style={{ color: "#8a9bb8" }}>{tokenAmount} × ?{fmt(selectedProperty.pricePerToken)}</span>
                    <span style={{ color: "#e8dcc8", fontWeight: 700 }}>? {fmt(tokenAmount * selectedProperty.pricePerToken)}</span>
                  </div>
                </div>
                <div style={{ marginBottom: 20 }}>
                  <label style={{ fontSize: 13, color: "#8a9bb8", display: "block", marginBottom: 8 }}>{t.payment.method}</label>
                  <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 10 }}>
                    {t.payment.methods.map((m, i) => (
                      <div key={i} onClick={() => setPaymentMethod(i)} style={{ padding: "10px 14px", borderRadius: 10, border: 1.5px solid ${paymentMethod === i ? "#c9a84c" : "rgba(201,168,76,0.15)"}, background: paymentMethod === i ? "rgba(201,168,76,0.1)" : "transparent", cursor: "pointer", fontSize: 13, color: paymentMethod === i ? "#c9a84c" : "#8a9bb8", transition: "all 0.2s" }}>{m}</div>
                    ))}
                  </div>
                </div>
                <div style={{ display: "flex", gap: 12 }}>
                  <button className="outline-btn" style={{ flex: 1, fontFamily: "inherit" }} onClick={() => setShowPayment(false)}>{isUrdu ? "?????" : "Cancel"}</button>
                  <button className="gold-btn" style={{ flex: 2, fontFamily: "inherit" }} onClick={handleConfirmPayment}>{t.payment.confirm}</button>
                </div>
              </>
            )}
          </div>
        </div>
      )}
    </div>
  );
}
